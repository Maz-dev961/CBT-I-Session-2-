                    Calculate Your Sleep Efficiency

                </h3>

                <p class="text-slate-600 mb-6">This is the percentage of time in bed that you were actually asleep. It's a key metric for our work.</p>

                <div class="bg-blue-50 rounded-lg p-6 text-center mb-8 border border-blue-100">

                    <p class="text-lg font-semibold text-blue-800">

                        Sleep Efficiency = (Total Sleep Time ÷ Time in Bed) × 100

                    </p>

                    <p class="text-slate-500 mt-2">Example: 6 hours slept ÷ 8 hours in bed = 0.75, or 75% efficiency.</p>

                </div>

        document.addEventListener('DOMContentLoaded', () => {

            const homeworkItems = document.querySelectorAll('.homework-item');

            const progressBar = document.getElementById('progress-bar');

            const progressText = document.getElementById('progress-text');

            const totalItems = homeworkItems.length;


            const updateProgress = () => {

                const completedItems = document.querySelectorAll('.homework-item.completed').length;

                const percentage = totalItems > 0 ? (completedItems / totalItems) * 100 : 0;

                

                progressBar.style.width = `${percentage}%`;

                progressText.textContent = `${Math.round(percentage)}%`;

            };


            homeworkItems.forEach(item => {

                item.addEventListener('click', () => {

                    item.classList.toggle('completed');

                    updateProgress();

                });

            });


            // Initial call to set progress on load

            updateProgress();

        });

    </script>

</body>

</html>
